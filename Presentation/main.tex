% \pipelinehighlight{N}: Track 1 only; box N full opacity, others faded to 0.2
\newcommand{\pipelinehighlight}[1]{%
  \pgfmathsetmacro{\opA}{(#1==1)?1.0:0.2}%
  \pgfmathsetmacro{\opB}{(#1==2)?1.0:0.2}%
  \pgfmathsetmacro{\opC}{(#1==3)?1.0:0.2}%
  \pgfmathsetmacro{\opD}{(#1==4)?1.0:0.2}%
  \pgfmathsetmacro{\arAB}{min(\opA,\opB)}%
  \pgfmathsetmacro{\arBC}{min(\opB,\opC)}%
  \pgfmathsetmacro{\arCD}{min(\opC,\opD)}%
  \centering
  \begin{tikzpicture}[node distance=0cm]
    \begin{scope}[opacity=\opA]
      \node[pipeboxA] (dp) {Data\\Preparation};
    \end{scope}
    \begin{scope}[opacity=\opB]
      \node[pipeboxB, right=0.7cm of dp] (im) {Interpretable\\Modeling};
    \end{scope}
    \begin{scope}[opacity=\opC]
      \node[pipeboxC, right=0.7cm of im] (fe) {Feature Effects\\[-1pt]\& Importance};
    \end{scope}
    \begin{scope}[opacity=\opD]
      \node[pipeboxD, right=0.7cm of fe] (fs) {Failure/Success\\Mode Analysis};
    \end{scope}
    \draw[pipearrow, opacity=\arAB] (dp.east)  -- (im.west);
    \draw[pipearrow, opacity=\arBC] (im.east)  -- (fe.west);
    \draw[pipearrow, opacity=\arCD] (fe.east)  -- (fs.west);
  \end{tikzpicture}%
}

    \node[pipeboxD, right=0.7cm of fe1] (fs1) {Failure/Success\\Mode Analysis};
