%--- Slide: Why interpretable models? ------------------------------------------
\begin{frame}{Why Interpretable Models?}
  \begin{itemize}[<+->]
    \item Per-trajectory \texttt{ml\_ade} alone do not tell us \emph{why} some trajectories are hard
    \item Interpretable models let us directly read off feature effects: how does speed, heading change or scene density affect prediction error?
    \item Two tested approaches:
    \begin{itemize}
        \item \textbf{GAM:} smooth effects directly readable from the model
        \item \textbf{XGBoost:} post-hoc via SHAP values and partial
          dependence plots
    \end{itemize}
    \item This turns ADE into \textcolor{lmugreen}{\textbf{actionable understanding}} of model
          behaviour
  \end{itemize}
\end{frame}

%--- Slide: GAM and XGBoost ---------------------------------------------------
\begin{frame}{GAM and XGBoost}
  \begin{columns}[t]
    \begin{column}{0.48\textwidth}
      \textbf{GAM (Generalized Additive Model)}\\
      \textit{\footnotesize Inherently interpretable}
      \visible<2->{\begin{itemize}[<+->]
        \item Models target as a sum of smooth functions: $f(x) = \sum_j s_j(x_j)$
        \item One spline per feature — no interaction terms
        \item p-values for significance testing
        \item Feature effects directly readable from the model
      \end{itemize}}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{XGBoost}\\
      \textit{\footnotesize Black-box, interpreted post-hoc}
      \visible<3->{\begin{itemize}[<+->]
        \item Gradient-boosted decision trees
        \item Flexibly captures non-linearities
              and feature interactions
        \item Interpreted via \textbf{SHAP values} and
              \textbf{partial dependence plots}
      \end{itemize}}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Validating the Tuning Procedure — Nested \\ Resampling}
  \begin{itemize}
    \item \textbf{Problem:} hyperparameter tuning risks overfitting to a
          particular data split
    \item \textbf{Solution:} nested cross-validation
    \begin{itemize}
      \item Outer folds: evaluate generalization of the full tuning procedure
      \item Inner folds: tune hyperparameters on each outer-fold training set via cross-validation
    \end{itemize}
    \item OOF evaluation: inspect out-of-fold predictions for instability
          and bias
  \end{itemize}
\end{frame}

%--- Return to validation slide with conclusion --------------------------------
\begin{frame}{Validating the Tuning Procedure — Nested Resampling}
  \begin{itemize}
    \item \textbf{Problem:} hyperparameter tuning risks overfitting to a
          particular data split
    \item \textbf{Solution:} nested cross-validation
    \begin{itemize}
      \item Outer folds: evaluate generalization of the full tuning procedure
      \item Inner folds: tune hyperparameters on each outer-fold training set via cross-validation
    \end{itemize}
    \item OOF evaluation: inspect out-of-fold predictions for instability
          and bias
    \item \textbf{Conclusion:} tuning generalises stably $\to$ proceed to
          full-data tuning + refit
  \end{itemize}
  \vspace{0.4em}
  \footnotesize Final models are refit on \emph{all} data with optimal
  hyperparameters — no held-out test set, because the goal is understanding,
  not out-of-fold performance.
\end{frame}

%--- Slide: GAM Variant Selection ----------------------------------------------
\begin{frame}{GAM Variant Selection}
  \begin{columns}[T]
    \begin{column}{0.44\textwidth}
      \textbf{Three candidates:}
      \begin{itemize}
        \item \textbf{LinearGAM (no log)} — fitted on raw \texttt{ml\_ade}
        \item \textbf{LinearGAM (log)} — fitted on
              $\log(1+\texttt{ml\_ade})$, evaluated on original scale
        \item \textbf{GammaGAM} — Gamma distribution with log link,
              suited for positive right-skewed targets
      \end{itemize}
      \vspace{0.5em}
      Selection criterion: lowest mean outer-fold RMSE (original scale).
    \end{column}
    \begin{column}{0.53\textwidth}
      \renewcommand{\arraystretch}{1.3}
      \begin{tabular}{lccc}
        \toprule
        \textbf{Variant} & \textbf{RMSE} & \textbf{R$^2$} & \textbf{MAE} \\
        \midrule
        LinearGAM & 0.448 & 0.491 & 0.263 \\
        \textcolor{lmugreen}{\textbf{LinearGAM (log)}} & \textcolor{lmugreen}{\textbf{0.445}} & \textcolor{lmugreen}{\textbf{0.498}} & \textcolor{lmugreen}{0.250} \\
        GammaGAM         & 0.445          & 0.498          & 0.253 \\
        \bottomrule
      \end{tabular}\\[0.4em]
      \footnotesize Outer-fold means, 3-fold nested CV\@.\\
      \textcolor{lmugreen}{\textbf{LinearGAM (log) selected.}}
    \end{column}
  \end{columns}
\end{frame}
