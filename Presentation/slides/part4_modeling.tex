%--- Slide: Why interpretable models? ------------------------------------------
\begin{frame}{Why Interpretable Models?}
  \begin{itemize}[<+->]
    \item Per-trajectory \texttt{ml\_ade} alone do not tell us \emph{why} some trajectories are hard
    \item Interpretable models let us directly read off feature effects: how does speed, heading change or scene density affect prediction error?
    \item Two tested approaches:
    \begin{itemize}[<3->]
        \item \textbf{GAM:} smooth effects directly readable from the model
        \item \textbf{XGBoost:} post-hoc via SHAP values and partial
          dependence plots
    \end{itemize}
    \item This turns ADE into \textcolor{lmugreen}{\textbf{actionable understanding}} of model
          behaviour
  \end{itemize}
\end{frame}

%--- Slide: GAM and XGBoost ---------------------------------------------------
\begin{frame}{GAM and XGBoost}
  \begin{columns}[t]
    \begin{column}{0.48\textwidth}
      \textbf{GAM (Generalized Additive Model)}\\
      \textit{\footnotesize Inherently interpretable}
      \visible<2->{\begin{itemize}
        \item Models target as a sum of smooth functions: $f(x) = \sum_j s_j(x_j)$
        \item One spline per feature — no interaction terms
        \item p-values for significance testing
        \item Feature effects directly readable from the model
      \end{itemize}}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{XGBoost}\\
      \textit{\footnotesize Black-box, interpreted post-hoc}
      \visible<3->{\begin{itemize}
        \item Gradient-boosted decision trees
        \item Flexibly captures non-linearities
              and feature interactions
        \item Interpreted via \textbf{SHAP values} and
              \textbf{partial dependence plots}
        \item SHAP: per-row, signed contribution of each feature,
              additive on top of a baseline (mean model prediction)
              : $$f(x) = \phi_0 + \sum_j \phi_j(x)$$      \end{itemize}}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}[t]{Validating Tuning Procedure - Nested Resampling}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.53\textwidth}
      \small
      \begin{itemize}
        \item<2-> \textbf{Problem:} hyperparameter tuning risks overfitting to a
              particular data split
        \item<3-> \textbf{Solution:} nested cross-validation
        \begin{itemize}
          \item<3-> Outer folds: evaluate generalization of the full tuning procedure
          \item<3-> Inner folds: tune hyperparameters on each outer-fold training set via cross-validation
        \end{itemize}
        \item<4-> OOF evaluation: inspect out-of-fold predictions for instability
              and bias
      \end{itemize}
    \end{column}

\begin{frame}[t]{Model Variant Selection}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.48\textwidth}
      \small
      \textbf{GAM variants:}
      \begin{itemize}\setlength\itemsep{0.15em}
        \item \textbf{LinearGAM (no log)} — fitted on raw \texttt{ml\_ade}
        \item \textbf{LinearGAM (log)} — fitted on
              $\log(1+\texttt{ml\_ade})$, evaluated on original scale
        \item \textbf{GammaGAM} — Gamma distribution with log link,
