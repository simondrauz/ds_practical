\section{Conclusions}

%--- Summary of Findings ------------------------------------------------------
\begin{frame}{Summary of Findings}
  \begin{enumerate}
    \item<1-> \textbf{RQ1 — Which characteristics drive prediction error?}
    \begin{itemize}
      \item \texttt{std\_speed} (speed variation) and
            \texttt{heading\_change\_per\_sec} (turning) dominate
      \item \texttt{max\_speed} contributes; \texttt{mean\_acceleration}
            is near-zero
      \item Scene-context features (density, vehicle count) carry little
            explanatory power for per-trajectory error
    \end{itemize}
    \item<2-> \textbf{RQ2 — What are the success and failure modes?}
    \begin{itemize}
      \item \emph{Success:} \emph{Slow turners (low speed bounds turning)}; \emph{Straight walkers (stable heading bounds speed)}
      \item \emph{Failure:} \emph{Erratic motion (speed variation + turning)}; \emph{Raw speed paired with turning}
    \end{itemize}
    \item<3-> \textbf{RQ3 — How do model settings affect performance?}
    \begin{itemize}
      \item \texttt{prediction\_sec} has the largest effect — longer
            horizon raises error strongly
      \item \texttt{attention\_radius\_m} second — short radius is
            beneficial
      \item \texttt{history\_sec} has overall little impact in the swept range
    \end{itemize}
  \end{enumerate}
\end{frame}

%--- Limitations --------------------------------------------------------------
\begin{frame}{Limitations}
  \begin{itemize}[<+->]
    \item \textbf{Meta-models explain Trajectron++, not ground truth}
      \begin{itemize}[<1->]
        \item Drivers are inferred via meta-models as proxies —
              Trajectron++ does not expose them directly
      \end{itemize}
    \item \textbf{Meta-model under-prediction at high ADE}
      \begin{itemize}[<2->]
        \item OOF diagnostics: strong agreement at low/moderate ADE,
              systematic under-prediction in the high-error tail
        \item Implication: trajectory + scene features do not cover
              what makes the worst cases worst
      \end{itemize}
    \item \textbf{Scope}
      \begin{itemize}[<3->]
        \item Pedestrian trajectories on nuScenes only — no transfer
              guarantee to other agent types or datasets
        \item Settings sweep uses nuScenes \emph{mini} — results may
              not transfer to full-scale training
      \end{itemize}
  \end{itemize}
\end{frame}

%--- Future Work --------------------------------------------------------------
\begin{frame}{Future Work}
  \begin{itemize}[<+->]
    \item \textbf{Pipeline as a tool — model- and dataset-agnostic by design}
      \begin{itemize}[<1->]
        \item Meta-model + feature-effect clustering is decoupled from
              the prediction model and the dataset
        \item Via the \texttt{trajdata} interface, same pipeline
              ports to other trajectory prediction models and datasets
              with little effort
      \end{itemize}
    \item \textbf{Richer input space}
      \begin{itemize}[<2->]
        \item Add map / lane geometry, location, finer social context
        \item Targets the under-prediction ceiling and may unlock
              cleaner hard-regime sub-structure
      \end{itemize}
    \item \textbf{Other agents}
      \begin{itemize}[<3->]
        \item Extend to vehicle, cyclist, and motorcyclist trajectories
      \end{itemize}
  \end{itemize}
\end{frame}
