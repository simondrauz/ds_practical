\begin{frame}{Target Distribution \& Log Transformation}
  \begin{columns}
    \begin{column}{0.50\textwidth}
      \begin{itemize}[<+->]
        \item \texttt{ml\_ade} is strongly \emph{right-skewed}
              (skewness $= 2.59$)
        \item Small errors are very frequent; large errors are rare
              but influential
        \item Log transformation (\texttt{log1p}) applied to stabilise
              variance and improve model fit
        \item Models are trained on log-scale target,
              evaluated on original scale
      \end{itemize}

%--- Slide: Feature Selection — two approaches --------------------------------
\begin{frame}{Feature Selection — Two Approaches}
  \begin{columns}[t]
    \begin{column}{0.48\textwidth}
      \textbf{Approach 1: MI elbow + VIF}
      \begin{itemize}[<+->]
        \item Rank features by Mutual Information with target
        \item Drop low-signal tail below the elbow: 18 $\to$ 10
        \item Apply VIF filter (threshold $= 5$): 10 $\to$ \textbf{4 features}
      \end{itemize}
      \onslide<4->{
      \vspace{0.4em}
      \small
      \texttt{std\_speed}, \texttt{mean\_acceleration},\\
      \texttt{max\_speed}, \texttt{heading\_change\_per\_sec}\\[0.3em]
      \footnotesize Motion features only
      ($p \ll 0.01$).
      }
    \end{column}
    \begin{column}{0.48\textwidth}
      \onslide<5->{\textbf{Approach 2: VIF only}}
      \begin{itemize}
        \item<5-> Skip MI; apply VIF directly to all 18 candidates
        \item<5-> VIF filter (threshold $= 5$): 18 $\to$ \textbf{6 features}
      \end{itemize}
      \onslide<6->{
      \vspace{0.4em}
      \small
      \texttt{std\_speed}, \texttt{mean\_acceleration},\\
      \texttt{min\_neighbor\_distance},
      \texttt{heading\_change\_per\_sec},\\
      \texttt{scene\_num\_VEHICLE},
      \texttt{scene\_density\_VEHICLE}\\[0.3em]
      \footnotesize Retains scene-level features alongside motion features.
      }
    \end{column}
  \end{columns}
  \vspace{0.5em}
\end{frame}
