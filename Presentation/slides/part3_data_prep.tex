%--- Slide: Feature Selection via Mutual Information --------------------------
\begin{frame}{Feature Selection via Mutual Information}
  \begin{columns}
    \begin{column}{0.52\textwidth}
      \begin{itemize}[<+->]
        \item Features ranked by Mutual Information (MI) with target
        \item MI captures \emph{non-linear} dependencies, unlike correlation
        \item Elbow detection: largest single-step drop in ranked MI curve
              separates informative features from low-signal tail
        \item Features below the elbow are dropped
        \item \textbf{18 $\to$ 10 candidate features}
      \end{itemize}
    \end{column}

%--- Slide: Multicollinearity Filter (VIF) ------------------------------------
\begin{frame}{Handling Multicollinearity with VIF}
  \begin{columns}
    \begin{column}{0.52\textwidth}
      \begin{itemize}[<+->]
        \item Collinear features split SHAP attribution arbitrarily and
              make GAM smooth terms unidentifiable
        \item Iterative VIF elimination (threshold $= 5$)
        \item At each step: drop the feature above threshold with the
              \emph{lowest} $|\rho_\text{Spearman}|$ with target
        \item Model settings always protected
      \end{itemize}
      \vspace{0.4em}
    \end{column}
    \begin{column}{0.45\textwidth}
      \onslide<3->{
      \textbf{Dropped (VIF $> 5$):}\\[0.3em]
      \small
      \begin{tabular}{lr}
        \texttt{mean\_speed}      & VIF $= 357$ \\
        \texttt{max\_jerk}        & VIF $= 20$  \\
        \texttt{mean\_jerk}       & VIF $= 12$  \\
        \texttt{max\_acceleration}& VIF $= 10$  \\
        \texttt{path\_efficiency} & VIF $= 9$   \\
      \end{tabular}
      }
    \end{column}
  \end{columns}
        \onslide<4->{
      \center{\textbf{6 final features:}\\[0.3em]
      \small
      \texttt{std\_speed}, \texttt{mean\_acceleration},
      \texttt{min\_neighbor\_distance},\\
      \texttt{heading\_change\_per\_sec},
      \texttt{scene\_num\_VEHICLE},
      \texttt{scene\_density\_VEHICLE}}
      }
\end{frame}
