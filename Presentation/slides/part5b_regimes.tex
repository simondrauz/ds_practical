%--- Methodology --------------------------------------------------------------
\begin{frame}{Performance Group Methodology}
  \begin{enumerate}
    \item Performance groups via k-means on $\log(1+\texttt{ml\_ade})$
          $\to$ easy / medium / hard
    \item Per group: cluster the signed feature-effect vector
          \begin{itemize}
            \item Features: \texttt{std\_speed}, \texttt{max\_speed},
                  \texttt{heading\_change\_per\_sec}, \texttt{mean\_acceleration}
            \item Effect sources: \textbf{XGBoost SHAP} \textbar\
                  \textbf{GAM additive effects}
            \item Algorithms: \textbf{OPTICS} \textbar\ \textbf{HDBSCAN}
          \end{itemize}
    \item Select per group via \textbf{DBCV} + noise share +
          semantic quality
          $\Rightarrow$ OPTICS / XGBoost preferred everywhere
    \item Map clusters $\to$ raw trajectory characteristics
          $\to$ \textbf{success / failure modes}
  \end{enumerate}
  \vfill
  \footnotesize
  Positive effect $\to$ pushes \texttt{ml\_ade} \emph{up}; negative $\to$ down.
\end{frame}

%--- Easy ---------------------------------------------------------------------
\begin{frame}{Easy Regime — Success Modes}
  \footnotesize
  DBCV \textbf{0.41} \hspace{0.4em}\textbar\hspace{0.4em} 2 clusters
  \hspace{0.4em}\textbar\hspace{0.4em} group median \texttt{ml\_ade}
  \textbf{0.17\,m}
  \vspace{0.4em}
  \begin{itemize}\setlength\itemsep{0.1em}
    \item Two well-separated modes cover 70\% of easy
    \item Different motion profiles $\to$ same low-error outcome
  \end{itemize}
  \vspace{0.5em}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{Slow turners} \\
      \scriptsize 34\% of easy \textbar\ median \textbf{0.08\,m}
      \begin{itemize}\setlength\itemsep{0.15em}
        \item Very low max speed, very low speed variation
        \item High turning rate
        \item Turning $\uparrow$ error, but slow steady motion dominates
        \item \emph{Slow despite turning} $\Rightarrow$ easiest
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \onslide<2->{\textbf{Straight walkers}}\\
      \onslide<2->{\scriptsize 36\% of easy \textbar\ median \textbf{0.20\,m}}
      \begin{itemize}[<2->]\setlength\itemsep{0.15em}
        \item Stable heading, normal max speed
        \item Low speed variation
        \item Heading stability drives low predicted error
        \item \emph{Straight and steady} $\Rightarrow$ easy
      \end{itemize}
    \end{column}
  \end{columns}
  \vfill
  \onslide<3->{\textcolor{lmugreen}{\textbf{Takeaway:}}
  low speed \emph{or} stable heading $\to$ Trajectron++ stays on track.}
\end{frame}

%--- Medium -------------------------------------------------------------------
\begin{frame}{Medium Regime — Transition Zone}
  \footnotesize
  DBCV \textbf{0.40} \hspace{0.4em}\textbar\hspace{0.4em} 2 clusters
  \hspace{0.4em}\textbar\hspace{0.4em} group median \texttt{ml\_ade}
  \textbf{0.60\,m}
  \vspace{0.4em}
  \begin{itemize}\setlength\itemsep{0.1em}
    \item Transition zone between easy and hard
    \item One discrete mode (60\%) + continuous gradient toward hard
  \end{itemize}
  \vspace{0.5em}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{Ordinary medium}\\
      \scriptsize 60\% of medium \textbar\ median \textbf{0.55\,m}
      \begin{itemize}\setlength\itemsep{0.15em}
        \item Slightly elevated max speed
        \item Modest speed variation
        \item Heading near global median
        \item \texttt{max\_speed} = only consistent error driver
        \item Stable heading prevents slide into hard
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \onslide<2->{\textbf{Toward the hard boundary}}\\
      \onslide<2->{\scriptsize No second discrete mode}
      \begin{itemize}[<2->]\setlength\itemsep{0.15em}
        \item Higher error $\to$ speed variation $\uparrow$, turning $\uparrow$,
              acceleration / jerk $\uparrow$
        \item Same axes as hard, lower magnitudes
        \item Continuous gradient, not a stable second cluster
      \end{itemize}
    \end{column}
  \end{columns}
  \vfill
  \onslide<3->{\textcolor{lmugreen}{\textbf{Takeaway:}}
  speed pushes error up; stable heading keeps it bounded.}
\end{frame}

%--- Hard ---------------------------------------------------------------------
\begin{frame}{Hard Regime — Failure Modes}
  \footnotesize
  DBCV \textbf{0.09} \hspace{0.4em}\textbar\hspace{0.4em} 4 clusters
  \hspace{0.4em}\textbar\hspace{0.4em} group median \texttt{ml\_ade}
  \textbf{1.78\,m}
  \vspace{0.4em}
  \begin{itemize}\setlength\itemsep{0.1em}
    \item 99.5\% of hard rows: positive total feature effect
    \item Dominant axes: high speed variation + elevated turning
    \item Two distinct failure modes (25\% combined); rest spreads
          continuously along the same axes
  \end{itemize}
  \vspace{0.4em}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{Erratic motion at low pace}\\
      \scriptsize 8\% of hard \textbar\ median \textbf{1.6–1.9\,m}
      \begin{itemize}\setlength\itemsep{0.15em}
        \item Below-median pace
        \item Very high speed variation + elevated turning
        \item Both drivers raise error (different magnitudes of
              variation $\times$ turning)
        \item Mean acceleration well below average
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \onslide<2->{\textbf{Fast motion with mild turning}}\\
      \onslide<2->{\scriptsize 17\% of hard \textbar\ median \textbf{1.50\,m}}
      \begin{itemize}[<2->]\setlength\itemsep{0.15em}
        \item Only hard mode with above-median max speed
        \item Loss split between \texttt{max\_speed} and
              \texttt{heading\_change}
        \item Speed variation only mildly elevated
      \end{itemize}
    \end{column}
  \end{columns}
  \vfill
  \onslide<3->{\textcolor{lmugreen}{\textbf{Takeaway:}}
  failure is driven by \emph{erratic} motion, not raw speed —
  fast only matters when paired with turning.}
\end{frame}
