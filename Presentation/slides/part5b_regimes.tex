%--- Methodology (steps 2 & 3) ------------------------------------------------
\begin{frame}[t]{Performance Group Methodology}
  \begin{enumerate}
    \item Performance groups via k-means on $\log(1+\texttt{ml\_ade})$
          $\to$ easy / medium / hard
    \item Per group: cluster the signed feature-effect vector
          \begin{itemize}
            \item Features: \texttt{std\_speed}, \texttt{max\_speed},
                  \texttt{heading\_change\_per\_sec}, \texttt{mean\_acceleration}
            \item Effect sources: \textbf{XGBoost SHAP} \textbar\
                  \textbf{GAM additive effects}
            \item Algorithms: \textbf{OPTICS} \textbar\ \textbf{HDBSCAN}
          \end{itemize}
    \item<2-> Select per group via \textbf{DBCV} + noise share +
          semantic quality
          $\Rightarrow$ OPTICS / XGBoost preferred everywhere
          \begin{itemize}
            \item DBCV evaluates density-reachability and connectivity
                  within the data, $\in [-1, 1]$:
                  $+1$ = robust separation of dense, distinct groups;
                  $0$ = ambiguous structure, clusters no denser than
                  surrounding noise
          \end{itemize}
    \item<3-> Interpret clusters (explanation patterns) by grounding them in raw trajectory characteristics
          $\to$ \textbf{success / failure modes}
  \end{enumerate}
\end{frame}
