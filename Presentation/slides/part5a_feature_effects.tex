\begin{frame}{Feature Effects — \texttt{std\_speed} \& \texttt{heading\_change\_per\_sec}}
  \begin{columns}[T]
    \begin{column}{0.4\textwidth}
      \footnotesize
      \onslide<1->{%
        {\small\textbf{\texttt{std\_speed}} — strong, monotone positive}
        \begin{itemize}\setlength\itemsep{0.05em}
          \item More variation $\Rightarrow$ higher ADE, monotonically
          \item GAM confirms the trend within the PDP range
          \item[$\rightarrow$] \textit{Erratic speed is harder to forecast}
        \end{itemize}
      }
      \vspace{0.3em}
      \onslide<2->{%
        {\small\textbf{\texttt{heading\_change\_per\_sec}} — positive, non-linear}
        \begin{itemize}\setlength\itemsep{0.05em}
          \item Turning generally raises ADE; drop-off at high values
          \item GAM confirms the trend within the PDP range
          \item[$\rightarrow$] \textit{Turns hurt accuracy — but only up to a point}
        \end{itemize}
      }

%--- Slide 3: max_speed & mean_acceleration ----------------------------------
\begin{frame}{Feature Effects — \texttt{max\_speed} \& \texttt{mean\_acceleration}}
  \begin{columns}[T]
    \begin{column}{0.4\textwidth}
      \footnotesize
      \onslide<1->{%
        {\small\textbf{\texttt{max\_speed}} — positive, increasing}
        \begin{itemize}\setlength\itemsep{0.05em}
          \item Higher peak speeds raise predicted ADE
          \item GAM agrees; CI widens at the tail
          \item[$\rightarrow$] \textit{At the upper tail, both the data and the
                model's confidence run thin}
        \end{itemize}
      }
      \vspace{0.3em}
      \onslide<2->{%
        {\small\textbf{\texttt{mean\_acceleration}} — near-zero, mixed}
        \begin{itemize}\setlength\itemsep{0.05em}
          \item Oscillates around zero
          \item Population-level effect $\approx 0$, but signed
                per-trajectory effects still vary
          \item[$\rightarrow$] \textit{What matters more is that pedestrians are
                changing speed, not how quickly}
        \end{itemize}
      }
    \end{column}

%--- Slide 4: scene_density_VEHICLE ------------------------------------------
\begin{frame}{Feature Effects — \texttt{scene\_density\_VEHICLE}}
  \begin{columns}[T]
    \begin{column}{0.4\textwidth}
      \footnotesize
      {\small\textbf{\texttt{scene\_density\_VEHICLE}} — negligible, noisy}
      \begin{itemize}\setlength\itemsep{0.05em}
        \item lower density seems to increase predicted ADE, but PDP is noisy
        \item GAM smooth not significant — CI spans zero
        \item Other scene / social features
              (\texttt{scene\_num\_VEHICLE},
               \texttt{min\_neighbor\_distance})
              show similar near-flat PDPs
        \item[$\rightarrow$] \textit{Scene features do not robustly explain
                      individual trajectory error}
      \end{itemize}
