%--- Slide 2: std_speed & heading_change_per_sec ------------------------------
\begin{frame}{Feature Effects — \texttt{std\_speed} \& \texttt{heading\_change\_per\_sec}}
  \begin{columns}[T]
    \begin{column}{0.4\textwidth}
      \footnotesize
      \textbf{\texttt{std\_speed}} — strong, monotone positive
      \begin{itemize}\setlength\itemsep{0.05em}
        \item Top driver of predicted ADE
        \item More variation $\Rightarrow$ higher ADE, monotonically
        \item GAM confirms the trend within the PDP range
        \item \textit{Erratic speed is harder to forecast}
      \end{itemize}
      \vspace{0.3em}
      \textbf{\texttt{heading\_change\_per\_sec}} — positive, non-linear
      \begin{itemize}\setlength\itemsep{0.05em}
        \item Turning raises ADE; plateaus at high values
        \item GAM diverges near zero — slow pedestrians register
              large heading rates from small position jitter
        \item \textit{Turns hurt accuracy — but only up to a point}
      \end{itemize}

%--- Slide 3: max_speed & mean_acceleration ----------------------------------
\begin{frame}{Feature Effects — \texttt{max\_speed} \& \texttt{mean\_acceleration}}
  \begin{columns}[T]
    \begin{column}{0.4\textwidth}
      \footnotesize
      \textbf{\texttt{max\_speed}} — positive, broadly increasing
      \begin{itemize}\setlength\itemsep{0.05em}
        \item Higher peak speeds raise predicted ADE
        \item Second-strongest driver after \texttt{std\_speed}
        \item GAM agrees; CI widens at the tail (data sparsity)
        \item \textit{At the upper tail, both the data and the
              model's confidence run thin}
      \end{itemize}
      \vspace{0.3em}
      \textbf{\texttt{mean\_acceleration}} — near-zero, mixed
      \begin{itemize}\setlength\itemsep{0.05em}
        \item Oscillates around zero; GAM not significant
        \item Population-level effect $\approx 0$, but signed
              per-trajectory effects still vary
        \item \textit{What matters more is that pedestrians are
              changing speed (\texttt{std\_speed}), not how quickly}
      \end{itemize}
    \end{column}

\begin{frame}{Feature Effects — \texttt{scene\_density\_VEHICLE}}
  \begin{columns}[T]
    \begin{column}{0.4\textwidth}
      \footnotesize
      \textbf{\texttt{scene\_density\_VEHICLE}} — negligible, noisy
      \begin{itemize}\setlength\itemsep{0.05em}
        \item Noisy oscillation across the full range
        \item Effect scale negligible vs.\ motion features
        \item GAM smooth not significant — CI spans zero
        \item Other scene / social features
              (\texttt{scene\_num\_VEHICLE},
               \texttt{min\_neighbor\_distance})
              show similar near-flat PDPs
        \item \textit{Scene density does not robustly explain
                      individual trajectory error}
