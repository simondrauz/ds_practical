%--- Feature Effects: std_speed & heading_change_per_sec -----------------------
\begin{frame}{Feature Effects — \texttt{std\_speed} \& \texttt{heading\_change\_per\_sec}}
  \begin{columns}[T]
    \begin{column}{0.4\textwidth}
      \scriptsize
      \textbf{\texttt{std\_speed}} — strong, monotone positive
      \begin{itemize}\setlength\itemsep{0.1em}
        \item dominant predictor
        \item speed variability rises monotonically with predicted ADE
        \item GAM confirms behaviour within the PDP range
        \item \textit{Erratic speed changes are fundamentally harder to forecast.}
      \end{itemize}
      \vspace{0.5em}
      \textbf{\texttt{heading\_change\_per\_sec}} — positive, non-linear
      \begin{itemize}\setlength\itemsep{0.1em}
        \item turning raises ADE; plateaus at extreme values
        \item GAM diverges: high effect near zero, \emph{decreasing} with more turning
        \item \textit{Near-stationary agents inflate the low-turning tail; PDP trend is more robust.}
      \end{itemize}
      \vspace*{\fill}
      \scriptsize\textit{PDP: 5--95\% percentile range. GAM smooth: full feature range with 95\% CI.}
    \end{column}

%--- Feature Effects: max_speed & mean_acceleration ----------------------------
\begin{frame}{Feature Effects — \texttt{max\_speed} \& \texttt{mean\_acceleration}}
  \begin{columns}[T]
    \begin{column}{0.4\textwidth}
      \scriptsize
      \textbf{\texttt{max\_speed}} — positive, broadly increasing
      \begin{itemize}\setlength\itemsep{0.1em}
        \item higher peak speeds raise predicted ADE
        \item second-strongest driver after \texttt{std\_speed}
        \item GAM confirms; confidence bands widen at tail (data sparsity)
        \item \textit{Faster peak speeds widen the trajectory envelope.}
      \end{itemize}
      \vspace{0.5em}
      \textbf{\texttt{mean\_acceleration}} — near-zero, mixed
      \begin{itemize}\setlength\itemsep{0.1em}
        \item oscillates around zero across its full range
        \item GAM: not significant; consistent with XGBoost
        \item \textit{Acceleration alone does not drive prediction error.}
      \end{itemize}
      \vspace*{\fill}
      \scriptsize\textit{PDP: 5--95\% percentile range. GAM smooth: full feature range with 95\% CI.}
