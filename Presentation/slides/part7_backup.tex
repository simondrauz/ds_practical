%--- SHAP explanation ---------------------------------------------------------
\begin{frame}[noframenumbering]{SHAP Values - SHapley Additive exPlanations}
  \footnotesize
  For each trajectory $i$, SHAP decomposes the XGBoost prediction into
  per-feature contributions — quantifying how much each feature
  shifts $\log(1+\widehat{\texttt{ml\_ade}}_i)$ away from the global mean:
  \vspace{0.2em}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \textbf{Additive decomposition}
      \[
        f(\mathbf{x}_i) \;=\;
        \underbrace{E\bigl[f(\mathbf{x})\bigr]}_{\text{baseline}}
        \;+\; \sum_{j=1}^{p} \phi_{i,j}
      \]
      \vspace{0.1em}
      \textbf{Shapley value for feature $j$}
      \[
        \phi_j(\mathbf{x}) =
        \!\!\!\sum_{S\,\subseteq\, F\setminus\{j\}}
        \frac{|S|\,!\;(|F|-|S|-1)!}{|F|!}
        \bigl[
          f_{S\cup\{j\}}(\mathbf{x}_{S\cup\{j\}}) - f_S(\mathbf{x}_S)
        \bigr]
      \]
      \vspace{-0.3em}
      \scriptsize Averages the marginal contribution of $j$ over all
      possible feature subsets.
    \end{column}
    \begin{column}{0.45\textwidth}
      \textbf{Notation}
      \begin{itemize}\setlength\itemsep{0.18em}
        \item $\phi_{i,j}$: contribution of feature $j$ to trajectory $i$
        \item $F$: full feature set (trajectory \& scene features)
        \item $S \subseteq F\setminus\{j\}$: subset excluding $j$
        \item $f_S(\mathbf{x}_S)$: prediction using only features in $S$
        \item $E[f(\mathbf{x})]$: mean prediction across all trajectories
      \end{itemize}
      \vspace{0.4em}
      \textbf{Interpretation}\\[0.2em]
      $\phi_{i,j}>0$: feature $j$ pushes predicted ADE \textbf{up}\\
      $\phi_{i,j}<0$: feature $j$ pushes predicted ADE \textbf{down}\\[0.3em]
      Mean $|\phi_j|$ over all $i$ $\;\rightarrow\;$
      \textcolor{lmugreen}{\textbf{global feature importance}}
    \end{column}
  \end{columns}
\end{frame}

%--- DBCV: high-level interpretation -----------------------------------------
\begin{frame}[noframenumbering]{DBCV — Density-Based Clustering Validation}
  \footnotesize
  Density-Based Clustering Validation: internal score for non-spherical,
  density-based clusters (Moulavi et al., 2014).
  \vspace{0.45em}

  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \scriptsize
      \textbf{Core idea}
      \begin{itemize}\setlength\itemsep{0.25em}
        \item \textbf{Reward:} dense, connected clusters.
        \item \textbf{Reward:} low-density gaps between clusters.
        \item \textbf{Penalize:} sparse bridges and weak internal links.
        \item \textbf{Penalize:} high noise share via the final weighting.
      \end{itemize}
      \vspace{0.45em}
      \textbf{Validity per cluster}
      \[
        V(C_i)=
        \frac{\min_{j\neq i}\text{DSPC}(C_i,C_j)-\text{DSC}(C_i)}
             {\max\!\bigl(\min_{j\neq i}\text{DSPC}(C_i,C_j),\,
                          \text{DSC}(C_i)\bigr)}
      \]
      \vspace{-0.6em}
      \begin{itemize}\setlength\itemsep{0.15em}
        \item $\text{DSC}$: sparsest internal cluster link.
        \item $\text{DSPC}$: nearest density separation to another cluster.
      \end{itemize}
    \end{column}

    \begin{column}{0.48\textwidth}
      \scriptsize
      \textbf{Overall score}
      \[
        \text{DBCV} =
        \sum_{i=1}^{K} \frac{|C_i|}{|\mathcal{O}|}\, V(C_i)
      \]
      Noise has no own validity term, but remains in $\mathcal{O}$.
      \vspace{0.45em}
      
      \textbf{How to read it}
      \begin{itemize}\setlength\itemsep{0.15em}
        \item Range: $[-1,1]$; higher is better.
        \item Positive: dense clusters, lower-density gaps.
        \item Near $0$: ambiguous density structure.
        \item Negative: implausible partition.
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}
